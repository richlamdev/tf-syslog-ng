#!/bin/bash
yum install wget -y
subscription-manager repos --enable rhel-8-for-x86_64-supplementary-rpms
wget https://dl.fedoraproject.org/pub/epel/epel-release-latest-8.noarch.rpm
rpm -Uvh epel-release-latest-8.noarch.rpm

cd /etc/yum.repos.d/
wget https://copr.fedorainfracloud.org/coprs/czanik/syslog-ng336/repo/epel-8/czanik-syslog-ng335-epel-8.repo
yum install syslog-ng
systemctl enable syslog-ng
systemctl start syslog-ng
